# Operation: Start Transfer (Stream CSV to ADLS)

> **Action:** `startTransfer`
> **Backend:** [`api/mirror/index.js`](../api/mirror/index.js) — `startTransfer()`, `fetchQueueData()`, `streamBcToMirror()`
> **Frontend:** [`bc-open-mirror.html`](../../bc-open-mirror.html) — `runTableNow()` (Step 3)

---

## Purpose

Download the CSV data generated by BC queue tasks and stream it to the ADLS Gen2 landing
zone. This is a **fire-and-forget** operation: the backend starts the transfer in the
background and returns a `transferId` immediately. The frontend polls `checkTransferStatus`
to track progress.

## Request

```http
POST /api/mirror
Content-Type: application/json

{
  "action": "startTransfer",
  "companyId": "company-guid",
  "queueId": "queue-guid-1",
  "deletedQueueId": "queue-guid-2",
  "configId": "uuid-of-table-config"
}
```

## Response

```json
{
  "transferId": "uuid",
  "status": "running"
}
```

## Architecture: Fire-and-Forget Pattern

```javascript
async function startTransfer(conn, token, companyId, queueId, deletedQueueId, configId, lcid) {
  const transferId = crypto.randomUUID();

  // Persist running state BEFORE starting work
  await saveTransferState(conn, token, companyId, configId, {
    transferId, status: "running", createdAt: new Date().toISOString(),
  });

  // Track streaming progress in-memory
  _transferProgress[configId] = { totalBytes: 0, lineCount: 0 };
  const onProgress = ({ totalBytes, lineCount }) => {
    _transferProgress[configId] = { totalBytes, lineCount };
  };

  // Fire-and-forget: start download+upload in background
  fetchQueueData(conn, token, companyId, queueId, deletedQueueId, configId, lcid, onProgress)
    .then(async (result) => {
      delete _transferProgress[configId];
      await saveTransferState(conn, token, companyId, configId, {
        transferId, status: "completed", createdAt: new Date().toISOString(), result,
      });
    })
    .catch(async (err) => {
      delete _transferProgress[configId];
      await saveTransferState(conn, token, companyId, configId, {
        transferId, status: "error", createdAt: new Date().toISOString(), error: err.message,
      });
    });

  return { transferId, status: "running" };
}
```

## fetchQueueData: The Core Transfer Logic

This function orchestrates the full data download and upload:

### Step 1: Load Config and Queue Records

```javascript
const [connection, tableCfg] = await Promise.all([
  getMirrorConnection(conn, token, companyId),
  getTableConfig(conn, token, companyId, configId),
]);

// Fetch both queue records in parallel
const [queueRecord, deletedQueueRecord] = await Promise.all([
  queueId ? httpsJson(BC_HOST, `${queueBasePath}(${queueId})`, "GET", authHeaders, null)
           : Promise.resolve(null),
  deletedQueueId ? httpsJson(BC_HOST, `${queueBasePath}(${deletedQueueId})`, "GET", authHeaders, null)
                  : Promise.resolve(null),
]);
```

### Step 2: Validate Queue Results

Each queue result is validated. `datacontenttype` determines the outcome:

| `datacontenttype` | Has URL | Action |
|-------------------|---------|--------|
| `text/csv` | Yes | Stream to ADLS |
| *(empty)* | No | Skip (no records) |
| `text/json` | Yes | Auto-retry once, then error |

### Step 3: Stream to ADLS (4MB Chunks)

```javascript
const STREAM_CHUNK_SIZE = 4 * 1024 * 1024; // 4MB

function streamBcToMirror(bcHostname, bcPath, authHeaders, fileClient, onProgress) {
  return new Promise((resolve, reject) => {
    const req = https.request({
      hostname: bcHostname, path: bcPath, method: "GET", headers: authHeaders
    }, (res) => {
      let offset = 0;
      let lineCount = 0;
      let pending = Buffer.alloc(0);

      const processChunks = async () => {
        for await (const chunk of res) {
          // Count newlines for record tracking
          for (let i = 0; i < chunk.length; i++) {
            if (chunk[i] === 0x0A) lineCount++;
          }

          pending = Buffer.concat([pending, chunk]);

          // Flush when we have enough for an ADLS append
          while (pending.length >= STREAM_CHUNK_SIZE) {
            const toSend = pending.subarray(0, STREAM_CHUNK_SIZE);
            pending = Buffer.from(pending.subarray(STREAM_CHUNK_SIZE));
            await fileClient.append(toSend, offset, toSend.length);
            offset += toSend.length;
            if (onProgress) onProgress({ totalBytes: offset, lineCount });
          }
        }

        // Flush remaining data
        if (pending.length > 0) {
          await fileClient.append(pending, offset, pending.length);
          offset += pending.length;
        }

        // Commit the file
        await fileClient.flush(offset);
        resolve({ totalBytes: offset, lineCount });
      };

      processChunks();
    });
    req.end();
  });
}
```

Key properties:
- **Backpressure**: Stream pauses during ADLS appends.
- **Peak memory**: ~4MB regardless of file size.
- **Progress tracking**: `onProgress` callback reports bytes and line count.

const stamp = formatMirrorFileStamp(confirmedDt || new Date());

### Step 5: Timestamp Management

```javascript
// Only save sync timestamp when this is the FINAL chunk
if (continueFromRecordId) {
  logs.push("Skipping timestamp update — continuation chunk pending");
} else {
  if (confirmedIso) {
    await setIntegrationTimestamp(conn, token, companyId, tableCfg, confirmedIso);
  }
}
```

The `confirmedIso` timestamp comes from the BC queue record's `time` field — this is
the server-side timestamp of when BC captured the snapshot, preventing clock drift.

### Step 6: Continuation Handling

When BC returns `continueFromRecordId` (non-null, non-zero GUID):

```javascript
const NULL_GUID = '00000000-0000-0000-0000-000000000000';
const rawContinueId = effectiveCsvRecord?.continueFromRecordId || null;
const continueFromRecordId = (rawContinueId && rawContinueId !== NULL_GUID)
  ? rawContinueId : null;
```

The frontend detects this and re-triggers the scheduler after 10 seconds to process
the next chunk.

## Transfer State Persistence

Transfer state is stored in BC's `Cloud Events Storage` to survive across Azure Function
instances:

```javascript
// Per-process mutex prevents concurrent read-modify-write conflicts
let _transferStateMutex = Promise.resolve();
function withTransferLock(fn) {
  const prev = _transferStateMutex;
  let unlock;
  _transferStateMutex = new Promise(r => { unlock = r; });
  return prev.then(fn).finally(unlock);
}

async function saveTransferState(conn, token, companyId, configId, entry) {
  return withTransferLock(async () => {
    const state = await getTransferState(conn, token, companyId);
    state[configId] = entry;
    await setConfig(conn, token, companyId, CONFIG_TRANSFER_STATE_ID, JSON.stringify(state));
  });
}
```

## Progress Tracking

In-memory progress is tracked for the current Azure Function instance:

```javascript
const _transferProgress = {};

// Updated during streaming
_transferProgress[configId] = { totalBytes: 0, lineCount: 0 };
```

`checkTransferStatus` returns this progress for running transfers.

## Enterprise Upgrade Path

In the [enterprise architecture](../ENTERPRISE-OPEN-MIRROR-ARCHITECTURE.md):

- Transfer is a **Durable Functions activity** managed by the orchestrator.
- Streaming happens in a dedicated **Worker** (not the API function).
- Progress is tracked in Durable Functions entity state.
- Multi-zone: same CSV is streamed to multiple landing zones in parallel.
- Retry with checkpoint: if a stream fails mid-way, resume from the last committed offset.
- Large file support: files > 4GB use ADLS block blob with committed block lists.
